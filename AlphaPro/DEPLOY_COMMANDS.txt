# AlphaPro Docker Deployment Commands

# Make sure Docker Desktop is running, then run:

cd AlphaPro

# Clean up old containers and build fresh
docker-compose -p alphapro down -v
docker-compose -p alphapro build --no-cache
docker-compose -p alphapro up -d

# Verify it's running
docker ps

# Check logs
docker-compose -p alphapro logs -f

# Test API
curl http://localhost:3001/api/health
curl http://localhost:3001/api/engine/state
